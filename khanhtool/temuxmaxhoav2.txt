#!/bin/bash

VERSION="4.68"
BASE_URL="https://raw.githubusercontent.com/khanhitxiaomi/khanhitxiaomi/main/khanhxiaomi"

set +x 2>/dev/null

RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
WHITE='\033[1;37m'
NC='\033[0m'

# ==============================
# HIỆU ỨNG LOADING
# ==============================
loading() {
  echo -ne "${CYAN}⏳ Đang xử lý"
  for i in {1..3}; do
    echo -ne "."
    sleep 0.4
  done
  echo -e "${NC}"
}

# ==============================
# KIỂM TRA KEY
# ==============================
check_key() {
  DEVICE=$(getprop ro.serialno 2>/dev/null || hostname)

  while true; do
    clear

    echo ""
    echo -e "${CYAN}📺 CÔNG CỤ CÀI ĐẶT TIVI XIAOMI BẢN QUYỀN${NC}"
    echo -e "${YELLOW}⚡ TỰ ĐỘNG - NHANH - ỔN ĐỊNH - UPDATE${NC}"
    echo -e "${GREEN}👨‍💻 Khánh IT | 0967.341.608 | 4.68${NC}"
    echo ""

    echo -e "${BLUE}🔑 VUI LÒNG NHẬP MẬT KHẨU${NC}"
    echo -e "${WHITE}➤ Nhập đúng KEY để sử dụng tool${NC}"
    echo -e "${WHITE}➤ KEY phân biệt chữ HOA / chữ thường${NC}"
    echo -e "${WHITE}➤ Nhập sai sẽ yêu cầu nhập lại${NC}"
    echo ""
    echo -e "${WHITE}➤ KEY DÙNG THỬ: 123${NC}"
    echo ""
    read -p "👉 Nhập KEY: " USER_KEY
    USER_KEY=$(echo "$USER_KEY" | xargs)

    if [ -z "$USER_KEY" ]; then
      echo -e "${RED}❌ Bạn chưa nhập KEY!${NC}"
      sleep 1.5
      continue
    fi

    loading

    DATA=$(curl -s --connect-timeout 5 "$BASE_URL/keys.txt")

    if [ -z "$DATA" ]; then
      echo -e "${RED}❌ Lỗi kết nối server!${NC}"
      sleep 2
      continue
    fi

    LINE=$(echo "$DATA" | grep -m 1 "^${USER_KEY}|")

    if [ -z "$LINE" ]; then
      echo -e "${RED}❌ KEY KHÔNG ĐÚNG!${NC}"
      echo -e "${YELLOW}👉 Vui lòng nhập lại...${NC}"
      sleep 1.5
      continue
    fi

    STATUS=$(echo "$LINE" | cut -d'|' -f2)
    KEY_DEVICE=$(echo "$LINE" | cut -d'|' -f3)

    if [ "$STATUS" != "active" ]; then
      echo -e "${RED}❌ KEY ĐÃ BỊ KHÓA!${NC}"
      sleep 1.5
      continue
    fi

    if [ -n "$KEY_DEVICE" ] && [ "$KEY_DEVICE" != "$DEVICE" ]; then
      echo -e "${RED}❌ KEY KHÔNG DÙNG CHO THIẾT BỊ NÀY!${NC}"
      sleep 1.5
      continue
    fi

    echo ""
    echo -e "${GREEN}=============================="
    echo -e "   🎉 XÁC THỰC THÀNH CÔNG"
    echo -e "==============================${NC}"
    sleep 1
    break
  done
}

# ==============================
# KIỂM TRA CẬP NHẬT
# ==============================
check_update() {
  NEW_VERSION=$(curl -s --connect-timeout 5 "$BASE_URL/version.txt")

  if [ -z "$NEW_VERSION" ]; then
    echo -e "${RED}❌ Không kiểm tra được update${NC}"
    return
  fi

  if [ "$NEW_VERSION" != "$VERSION" ]; then
    echo -e "${YELLOW}🚀 ĐANG CẬP NHẬT TOOL...${NC}"
    loading
    curl -s -o tool.sh "$BASE_URL/tool.sh"
    chmod +x tool.sh
    echo -e "${GREEN}✅ ĐÃ CẬP NHẬT - MỞ LẠI TOOL${NC}"
    exit 0
  fi
}

# ==============================
# LOAD CORE
# ==============================
load_core() {
  echo -e "${BLUE}📦 ĐANG LOAD CORE...${NC}"

  TMP="/sdcard/Download/core.txt"

  loading

  curl -s --connect-timeout 10 "$BASE_URL/core.txt" -o "$TMP"

  if [ ! -s "$TMP" ]; then
    echo -e "${RED}❌ LOAD CORE THẤT BẠI${NC}"
    exit 1
  fi

  CODE=$(base64 -d "$TMP" 2>/dev/null)

  if [ -z "$CODE" ]; then
    echo -e "${RED}❌ GIẢI MÃ LỖI${NC}"
    exit 1
  fi

  echo -e "${GREEN}🚀 KHỞI ĐỘNG TOOL...${NC}"
  sleep 1

  bash -c "$CODE"
}

# ==============================
# RUN
# ==============================
clear
echo ""
echo -e "${GREEN}🚀 ĐÂY LÀ TOOL TIVI XIAOMI BẢN QUYỀN${NC}"
echo -e "${CYAN}📺 Công cụ cài đặt & tối ưu Tivi Xiaomi${NC}"

echo ""
echo -e "${YELLOW}⚡ Tự động hoá toàn bộ: cài app • cấp quyền • tối ưu hệ thống${NC}"
echo -e "${WHITE}🔧 Hỗ trợ: ADB • APK • Android TV • TV nội địa / quốc tế${NC}"

echo ""
echo -e "${GREEN}👨‍💻 Khánh IT${NC} ${WHITE}| ☎ 0967.341.608${NC} ${WHITE}| 🔢 Phiên bản: 4.68${NC}"

echo ""
echo -e "${BLUE}📡 Trạng thái: Đang khởi động tool...${NC}"
echo -e "${CYAN}⏳ Vui lòng chờ trong giây lát${NC}"
echo ""


check_key
check_update
load_core
