#!/bin/bash

VERSION="4.68"
BASE_URL="https://raw.githubusercontent.com/khanhitxiaomi/khanhitxiaomi/main/khanhxiaomi"

set +x 2>/dev/null

RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
NC='\033[0m'

# ==============================
# CHECK KEY UI TV STYLE (FIXED)
# ==============================
check_key() {
  DEVICE=$(getprop ro.serialno 2>/dev/null || hostname)

  while true; do
    clear

    echo -e "${BLUE}╔════════════════════════════╗${NC}"
    echo -e "${GREEN}      XIAOMI TV TOOL         ${NC}"
    echo -e "${BLUE}╚════════════════════════════╝${NC}"
    echo ""

    echo -e "${CYAN}📺 DEVICE : ${YELLOW}$DEVICE${NC}"
    echo ""

    echo -ne "${YELLOW}🔑 ENTER KEY: ${NC}"
    read -r -p USER_KEY
    USER_KEY=$(echo "$USER_KEY" | xargs)

    echo ""
    echo -e "${BLUE}──────────────────────────────${NC}"
    echo -e "${CYAN}INPUT RECEIVED:${NC}"
    echo -e "${GREEN}$USER_KEY${NC}"
    echo -e "${BLUE}──────────────────────────────${NC}"
    echo ""

    echo -e "${BLUE}⏳ VERIFYING...${NC}"
    sleep 0.5

    # get key list
    DATA=$(curl -s --connect-timeout 5 "$BASE_URL/keys.txt")

    if [ -z "$USER_KEY" ]; then
      echo -e "${RED}❌ EMPTY KEY${NC}"
      sleep 1
      continue
    fi

    LINE=$(echo "$DATA" | grep -m 1 "^${USER_KEY}|")

    if [ -z "$LINE" ]; then
      echo -e "${RED}❌ WRONG KEY${NC}"
      sleep 1
      continue
    fi

    STATUS=$(echo "$LINE" | cut -d'|' -f2)
    KEY_DEVICE=$(echo "$LINE" | cut -d'|' -f3)

    if [ "$STATUS" != "active" ]; then
      echo -e "${RED}❌ KEY BLOCKED${NC}"
      sleep 1
      continue
    fi

    if [ -n "$KEY_DEVICE" ] && [ "$KEY_DEVICE" != "$DEVICE" ]; then
      echo -e "${RED}❌ DEVICE NOT MATCH${NC}"
      sleep 1
      continue
    fi

    echo ""
    echo -e "${GREEN}=============================="
    echo "   CONNECT SUCCESSFUL"
    echo -e "==============================${NC}"
    sleep 1
    break
  done
}

# ==============================
# CHECK UPDATE
# ==============================
check_update() {
  NEW_VERSION=$(curl -s "$BASE_URL/version.txt")

  if [ "$NEW_VERSION" != "$VERSION" ]; then
    echo -e "${YELLOW}🚀 UPDATING TOOL...${NC}"
    curl -s -o tool.sh "$BASE_URL/tool.sh"
    chmod +x tool.sh
    echo -e "${GREEN}✅ UPDATED - REOPEN TOOL${NC}"
    exit 0
  fi
}

# ==============================
# LOAD CORE (BASE64 EXEC)
# ==============================
load_core() {
  echo -e "${BLUE}📦 LOADING CORE...${NC}"

  TMP="/sdcard/Download/core.txt"

  curl -s "$BASE_URL/core.txt" -o "$TMP"

  if [ ! -s "$TMP" ]; then
    echo -e "${RED}❌ LOAD CORE FAILED${NC}"
    exit 1
  fi

  CODE=$(base64 -d "$TMP" 2>/dev/null)

  if [ -z "$CODE" ]; then
    echo -e "${RED}❌ DECODE FAILED${NC}"
    exit 1
  fi

  bash -c "$CODE"
}

# ==============================
# RUN
# ==============================
clear
echo -e "${GREEN}=============================="
echo "     XIAOMI TV TOOL LOADER"
echo -e "==============================${NC}"

check_key
check_update
load_core
