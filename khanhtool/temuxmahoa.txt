#!/bin/bash

VERSION="4.68"
BASE_URL="https://raw.githubusercontent.com/khanhitxiaomi/khanhitxiaomi/main/khanhxiaomi"

set +x 2>/dev/null

RED='\033[1;31m'
GREEN='\033[1;32m'
YELLOW='\033[1;33m'
BLUE='\033[1;34m'
CYAN='\033[1;36m'
NC='\033[0m'

# ===== CHECK KEY (FIX LOOP) =====
check_key() {
  DEVICE=$(getprop ro.serialno 2>/dev/null || hostname)

  while true; do
    echo -e "${CYAN}🔐 NHẬP KEY:${NC} "
    read -r USER_KEY

    DATA=$(curl -s "$BASE_URL/keys.txt")

    # FIX: trim + tránh lỗi grep rỗng
    LINE=$(echo "$DATA" | grep -E "^${USER_KEY}\|" 2>/dev/null)

    if [ -z "$USER_KEY" ]; then
      echo -e "${YELLOW}⚠️ CHƯA NHẬP KEY${NC}"
      continue
    fi

    if [ -z "$LINE" ]; then
      echo -e "${RED}❌ KEY SAI - NHẬP LẠI${NC}"
      continue
    fi

    STATUS=$(echo "$LINE" | cut -d'|' -f2)
    KEY_DEVICE=$(echo "$LINE" | cut -d'|' -f3)

    if [ "$STATUS" != "active" ]; then
      echo -e "${RED}❌ KEY BỊ KHÓA${NC}"
      continue
    fi

    if [ -n "$KEY_DEVICE" ] && [ "$KEY_DEVICE" != "$DEVICE" ]; then
      echo -e "${RED}❌ SAI THIẾT BỊ${NC}"
      continue
    fi

    echo -e "${GREEN}✅ KEY HỢP LỆ${NC}"
    break
  done
}

# ===== UPDATE =====
check_update() {
  NEW_VERSION=$(curl -s "$BASE_URL/version.txt")

  if [ "$NEW_VERSION" != "$VERSION" ]; then
    echo -e "${YELLOW}🚀 UPDATE...${NC}"
    curl -s -o tool.sh "$BASE_URL/tool.sh"
    chmod +x tool.sh
    echo -e "${GREEN}✅ DONE - REOPEN TOOL${NC}"
    exit 0
  fi
}

# ===== LOAD CORE =====
load_core() {
  echo -e "${BLUE}📦 LOAD CORE...${NC}"

  TMP="/sdcard/Download/core.txt"

  curl -s "$BASE_URL/core.txt" -o "$TMP"

  if [ ! -s "$TMP" ]; then
    echo -e "${RED}❌ LOAD FAIL${NC}"
    exit 1
  fi

  CODE=$(base64 -d "$TMP" 2>/dev/null)

  if [ -z "$CODE" ]; then
    echo -e "${RED}❌ DECODE FAIL${NC}"
    exit 1
  fi

  bash -c "$CODE"
}

# ===== RUN =====
clear
echo -e "${GREEN}===== TOOL LOADER =====${NC}"

check_key
check_update
load_core
