# ===== THÔNG SỐ TIVI CHẠY TỰ DỘNG KHI KẾT NỐI THÀNH CÔNG =====
get_tv_info() {
  MODEL=$(adb shell getprop ro.product.model | tr -d '\r')
  BRAND=$(adb shell getprop ro.product.brand | tr -d '\r')
  PANEL=$(adb shell getprop ro.boot.mi.panel_size 2>/dev/null | tr -d '\r')
  ANDROID=$(adb shell getprop ro.build.version.release | tr -d '\r')
  PATCH=$(adb shell getprop ro.build.version.security_patch | tr -d '\r')
  BUILD_SHOW=$(adb shell getprop ro.build.version.incremental | tr -d '\r')

  [ -z "$PANEL" ] && PANEL="?"

  TV_INFO_LINE="${CYAN}📺 ${MODEL}${NC} | ${WHITE}${BRAND}${NC} | ${GREEN}${PANEL} Inch${NC}\n${MAGENTA}🤖 Android ${ANDROID}${NC} | ${BLUE}🏗️ ${BUILD_SHOW}${NC} | ${RED}🔒 ${PATCH}${NC}"
}

# ===== MENU KẾT NỐI TIVI CONNECT =====
menu_connect() {
  while true; do
    clear
    banner_menu1
    read -e -p $'\e[1;36m👉 Nhập IP (192.168.1.10): \e[0m' ip
    [ "$ip" = "0" ] && break

    adb connect "$ip:5555" >/dev/null 2>&1
    sleep 5

    if adb devices | grep -q "$ip:5555.*device"; then
      echo -e "${GREEN}✅ KẾT NỐI THÀNH CÔNG!${NC}"
      sleep 1
      get_tv_info
      menu_main
    else
      echo -e "${RED}❌ KẾT NỐI THẤT BẠI!${NC}"
      sleep 2
    fi
  done
}

# ===== MENU 2 =====
menu_main() {
  while true; do
    clear
    banner_menu2
    read -p "👉 Vui Lòng Chọn 0-8: " choice
    case $choice in
      1) download_all ;;
      2) download_all_pro ;;
      3) install_all ;;
      4) projectivy_install ;;
      5) cap_quyen ;;
      6) install_from_phone_download ;;
      7) pull_tv_download ;;
      8) reboot_tv ;;
      0) return ;;
      *) echo -e "${RED}❌ Sai lựa chọn${NC}"; sleep 1 ;;
    esac
  done
}

    "https://github.com/dinhkhackhanhdz-blip/khanhtool/releases/download/adb/projectiv1.plbackup"
  )

  clear
  echo -e "${CYAN}🚀 DOWNLOAD PRO MAX${NC}"
  echo "---------------------------------------"

  total=${#URLS[@]}
  ok=0
  fail=0
  count=0

  for url in "${URLS[@]}"; do
    file="$DIR/$(basename "$url")"
    ((count++))

    echo -e "${YELLOW}[$count/$total] 📦 $(basename "$url")${NC}"

    # ===== DOWNLOAD =====
    wget -q --show-progress --tries=2 --timeout=10 "$url" -O "$file"

    if [ $? -eq 0 ]; then
      echo -e "${GREEN}   ✅ OK${NC}"
      ((ok++))
    else
      echo -e "${RED}   ❌ FAIL${NC}"
      ((fail++))
    fi

    percent=$((count * 100 / total))
    echo -e "${CYAN}📊 Progress: $percent%${NC}"
    echo "---------------------------------------"
  done

  echo -e "${GREEN}🎉 HOÀN TẤT DOWNLOAD${NC}"
  echo -e "${CYAN}📦 Thành công: $ok | ❌ Lỗi: $fail${NC}"

  pause
}


# ===== CÀI GIAO DIỆN PROJECTIVY CHO TIVI QUỐC TẾ =====
install_all() {
  DIR="$HOME/storage/shared/apktool"

  echo -e "${CYAN}📲 BẮT ĐẦU CÀI ĐẶT GIAO DIỆN TIVI QUỐC TẾ...${NC}"
  echo "---------------------------------------"

  # ===== INSTALL APK =====
  for file in "$DIR"/*.apk; do
    if [ -f "$file" ]; then
      echo -e "${YELLOW}📦 $(basename "$file")${NC}"

      adb install -r -d "$file" >/dev/null 2>&1

      if [ $? -eq 0 ]; then
        echo -e "${GREEN}   ✅ OK${NC}"
      else
        echo -e "${RED}   ❌ FAIL${NC}"
      fi
    fi
  done

  echo "---------------------------------------"
  echo -e "${CYAN}⚙️ ĐANG CẤU HÌNH HỆ THỐNG...${NC}"

  # ===== SET LAUNCHER =====
  adb shell pm reset-preferred-activities
  adb shell cmd package set-home-activity com.spocky.projengmenu/.MainActivity
  adb shell am start -a android.intent.action.MAIN -c android.intent.category.HOME

  # ===== PERMISSIONS =====
  adb shell pm grant com.spocky.projengmenu android.permission.READ_EXTERNAL_STORAGE
  adb shell pm grant com.spocky.projengmenu android.permission.WRITE_SECURE_SETTINGS
  adb shell appwidget grantbind --package com.spocky.projengmenu --user 0

  # ===== KEYBOARD =====
  adb shell ime enable com.liskovsoft.leankeyboard/.LeanbackImeService
  adb shell ime set com.liskovsoft.leankeyboard/.LeanbackImeService

  # ===== VOICE =====
  adb shell cmd voiceinteraction set-interactor com.google.android.katniss/.service.HotwordService
  adb shell settings put secure voice_interaction_service com.google.android.katniss/.service.HotwordService

  echo "---------------------------------------"
  echo -e "${GREEN}🎉 HOÀN TẤT CÀI ĐẶT & CẤU HÌNH${NC}"
  echo -e "${CYAN}🚀 LAUNCHER ĐÃ ĐƯỢC KÍCH HOẠT${NC}"

  pause
}


# ===== CÀI GIAO DIỆN PROJECTIVY CHO TIVI NỘI ĐỊA =====
projectivy_install() {

  echo "=== AUTO INSTALL PROJECTIVY ==="

  # check adb
  adb get-state >/dev/null 2>&1 || {
    echo "❌ Chưa kết nối ADB"
    return
  }

  echo -e "${CYAN}📲 BẮT ĐẦU CÀI ĐẶT CÁC APP CẦN THIẾT...${NC}"
  echo "---------------------------------------"

  # ===== INSTALL APK =====
  for file in "$DIR"/*.apk; do
    if [ -f "$file" ]; then
      echo -e "${YELLOW}📦 $(basename "$file")${NC}"

      adb install -r -d "$file" >/dev/null 2>&1

      if [ $? -eq 0 ]; then
        echo -e "${GREEN}   ✅ OK${NC}"
      else
        echo -e "${RED}   ❌ FAIL${NC}"
      fi
    fi
  done


  # ===== CÀI ĐẶT GIAO DIỆN PROJECTIVY LAUNCHER =====
  echo "Đang cài giao diện mới và tắt giao diện CHINA..."
 
  printf "📦 com.spocky.projengmenu.apk ... "
  adb install -r "$DIR/com.spocky.projengmenu.apk" | tr -d '\n'
  echo " $( [ ${PIPESTATUS[0]} -eq 0 ] && echo "✅ OK" || echo "❌ Lỗi" )"

  printf "🚫 Disable com.mitv.tvhome ... "
  adb shell pm disable-user --user 0 com.mitv.tvhome && echo "✅ OK" || echo "❌ Lỗi"

  # ===== CONFIG =====
  echo "Cấu hình Giờ Việt Nam..."

  adb shell service call alarm 3 s16 Asia/Bangkok
  adb shell settings put global device_locales vi-VN
  adb shell settings put global sys_locale vi-VN
  adb shell settings put system system_locales vi-VN
  adb shell settings put global heads_up_notifications_enabled 0
  adb shell settings put global stay_on_while_plugged_in 3

  echo "Cấu hình Giao Diện Tiếng Việt..."
  adb shell pm grant com.spocky.projengmenu android.permission.READ_EXTERNAL_STORAGE
  adb shell pm grant com.spocky.projengmenu android.permission.WRITE_SECURE_SETTINGS
  adb shell am start -n com.spocky.projengmenu/com.spocky.projengmenu.ui.home.MainActivity

  echo "Cấu hình Bàn Phím Tiếng Việt..."
  adb shell ime enable com.liskovsoft.leankeyboard/.ime.LeanbackImeService
  adb shell ime set com.liskovsoft.leankeyboard/.ime.LeanbackImeService
  adb shell settings put secure default_input_method com.liskovsoft.leankeyboard/.ime.LeanbackImeService

  echo "Cấu hình Giộng Nói Tiếng Việt..."

  adb shell settings put secure voice_interaction_service com.google.android.katniss/.service.HotwordService

  # ===== REMOVE APP =====
  echo "ĐANG TẮT APP CHINA VÀ DỌN RÁC..."

  for pkg in \
    com.xiaomi.tvqs \
    com.mitv.gallery \
    com.xm.webcontent \
    com.mitv.toolhouse \
    com.miui.hybrid.tv \
    com.xiaomi.tweather \

    com.sohu.inputmethod.sogou.tv \

    com.xiaomi.mitv.karaoke.service \
    com.xiaomi.wfdsinkhelperservice \

    com.android.tv.settings
